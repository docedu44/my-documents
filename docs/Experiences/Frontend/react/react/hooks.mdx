export interface IPersonsProps {
  money: number;
  handleIncrease: () => void;
  name?: string;
}

const Person1: FC<IPersonsProps> = ({ money, handleIncrease, name }) => {
  return (
    <>
      <h3>{name} ${money}</h3>
      <button onClick={handleIncrease}>Increase</button>
    </>
  );
};

export default AuctionHoc(Person1);
